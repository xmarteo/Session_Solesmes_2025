	\begin{paracol}{2}

		label=\null, 
		leftmargin=0pt, 
		itemindent=10pt, 
		labelsep=0pt, 
		labelwidth=0pt, 
		rightmargin=0pt, 
		parsep=0pt, 
		itemsep=0pt,
		topsep=-2mm]
	\input{psalmi_la/#1_#2.tex}
	\end{itemize}
	\switchcolumn
	\begin{itemize}[
		label=\null, 
		leftmargin=0pt, 
		itemindent=10pt, 
		labelsep=0pt, 
		labelwidth=0pt, 
		rightmargin=0pt, 
		parsep=0pt, 
		itemsep=0pt,
		topsep=-2mm]
	\input{psalmi_fr/#1.tex}
	\end{itemize}
	\end{paracol}
}
